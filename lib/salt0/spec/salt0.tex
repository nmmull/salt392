\chapter*{Syntax}

\begin{align*}
  x
  &
  &\text{(variables, $\V$)} \\
  n
  &
  &\text{(integers, $\Z$)} \\
  e
  &\cceq
  \unitt
  \alt n
  \alt x
  &\text{(expressions, $\E$)} \\
  s
  &\cceq
  \lete x \eqe e
  &\text{(statements, $\mathcal S$)} \\
  p
  &\cceq
  e
  \alt s \semi p
  &\text{(programs, $\cP$)}
\end{align*}

\chapter*{Typing}

\begin{align*}
  t
  &\cceq
  \unitty
  \alt \intty
  &\text{(types, $\T$)} \\
  \\
  \G &\in \V \mapsto \T & \text{(contexts)}\\
  \\
  \G &\ent e : t &\text{(expressions)} \\
  \G &\ent s \dashv \G &\text{(statments)} \\
  \G &\ent p : t &\text{(programs)}
\end{align*}
\bigskip

\begin{mathpar}
  \inferrule*[right=unit]{
  }{
    \G \ent \unt : \unt
  }
  \and
  \inferrule*[right=int]{
    n \in \Z
  }{
    \G \ent n : \intt
  }
  \and
  \inferrule*[right=var]{
    (x \mapsto t) \in \G
  }{
    \G \ent x : t
  }
  \and
  \inferrule*[right=let]{
    x \not \in \dom(\G)
    \and
    \G \ent e : t
  }{
    \G \ent \lete x \eqe e \tne \G[x \mapsto t]
  }
  \and
  \inferrule*[right=prog]{
    \G_1 \ent s \dashv \G_2
    \and
    \G_2 \ent p : t
  }{
    \G_1 \ent s \semi p : t
  }

\end{mathpar}

\chapter*{Evaluation}

\newcommand{\config}[2]{\langle \ #1 \ , \ #2 \ \rangle}

\begin{align*}
  v
  &\cceq \unitt \alt n
  &\text{(values, $\Val$)} \\
  \\
  S &\in \V \mapsto \Val &\text{(store)} \\
  \\
  S &\ent e \evl v &\text{(expressions)} \\
  S &\ent s \tne S &\text{(statements)} \\
  S &\ent p \evl v &\text{(programs)}
\end{align*}
\bigskip

\begin{mathpar}
  \inferrule*[right=unit]{
  }{
    S \ent \unt \evl \unt
  }
  \and
  \inferrule*[right=int]{
    n \in \mathbb Z
  }{
    S \ent n \evl n
  }
  \and
  \inferrule*[right=var]{
    (x \mapsto v) \in S
  }{
    S \ent x \evl v
  }
  \and
  \inferrule*[right=let]{
    S \ent e \evl v
  }{
    S \ent \lete x \eqe e \tne {S[x \mapsto v]}
  }
  \and
  \inferrule*[right=prog]{
    S_1 \ent s \tne S_2
    \and
    S_2 \ent p \evl v
  }{
    S_1 \ent s \semi p \evl v
  }
\end{mathpar}
