\chapter*{Syntax}

\begin{align*}
  x
  &
  &\text{(variables, $\V$)} \\
  n
  &
  &\text{(integers, $\Z$)} \\
  w &\cceq x \alt \dref w &\text{(place expression, $\mathcal W$)} \\
  e
  &\cceq
  \unitt
  \alt n
  \alt w
  \alt \brw w
  \alt \mbrw w
  \alt \copye w
  \alt w \eqe e
  &\text{(expressions, $\E$)} \\
  s
  &\cceq
  \lete x \eqe e
  \alt \lete \mute x \eqe e
  \alt e
  &\text{(statements, $\mathcal S$)} \\
  p
  &\cceq
  e
  \alt s \semi p
  &\text{(programs, $\cP$)}
\end{align*}

\chapter*{Typing}

\begin{align*}
  t
  &\cceq
  \unitty
  \alt \intty
  \alt \brw w
  \alt \mbrw w
  &\text{(types, $\T$)} \\
  \tilde t
  &\cceq
  \mvty t
  \alt t
  &\text{(partial types, $\tilde \T$)} \\
  m &\cceq \mathsf{imm} \alt \mathsf{mut} &\text{(mutability)} \\
  u &\cceq \langle \tilde t \rangle^m &\text{(slot types, $\mathbb S_{\T}$)}\\
  \\
  \G &\in \V \mapsto \mathbb S_\T & \text{(contexts)}\\
  \\
  &\mathsf{copyable}(t) & \text{(copyability)} \\
  \G &\ent \mathsf{readable}(w) &\text{(readability)} \\
  \G &\ent \mathsf{writable}(w) &\text{(writability)} \\
  \G &\ent \tilde t \approx \tilde t &\text{(type compatibility)} \\
  \G &\ent w : u & \text{(place expressions)} \\
  \G &\ent e : t \tne \G &\text{(expressions)} \\
  \G &\ent s \tne \G &\text{(statments)} \\
  \G &\ent p : t \tne \G &\text{(programs)}
\end{align*}

    \G \ent \unt : \unt

  \and
  \inferrule*[right=int]{
    n \in \Z
  }{
    \G \ent n : \intt \tne \G
  }
  \and
  \inferrule*[right=readable]{
    \nexists y,l . (y \mapsto \mbrw \code *^l x) \in \G
  }{
    \readable \G {\code *^k x}
  }
  \and
  \inferrule*[right=copyUnit]{
  }{
    \copyable{\unt}
  }

  \inferrule*[right=copyInt]{

  }

  \inferrule*[right=copyRef]{

    \G \ent \brw w : \brw w \tne \G

    \writable \G w

    \mutable \G w
  }{
    \G \ent \mbrw w : \mbrw w \tne \G
  }
  \and
  \inferrule*[right=$\approx$-unit]{
  }{
    \G \ent \unt \approx \unt
  }
  \and
  \inferrule*[right=$\approx$-int]{

    \G \ent \intt \approx \intt
  }

    \G \ent w_2 : \langle \tilde t_2 \rangle^{m_2}

    \G \ent \brw w_1 \approx \brw w_2

\begin{align*}
  \writee(\G, x, t) &=
  \G[x \mapsto t] \\
  \writee(\G, \code *^{k + 1}x, t) &=
  \writee(\G, \code *^k w, t) \quad \text{where} \quad (x \mapsto \langle \mbrw w \rangle^m) \in \G \\
  \\
  \mathsf{replace}(\brw \code{*}^{k + 1}w_1 , w_1, \brw w_2) &= \brw \code *^k w_2 \\
  \mathsf{replace}(\brw \code{*}^{k + 1}w_1 , w_1, \mbrw w_2) &= \brw \code *^k w_2 \\
  \mathsf{replace}(\mbrw \code{*}^{k + 1}w_1 , w_1, \brw w_2) &= \mbrw \code *^k w_2 \\
  \mathsf{replace}(\mbrw \code{*}^{k + 1}w_1 , w_1, \mbrw w_2) &= \mbrw \code *^k w_2 \\
  \mathsf{replace}(t_1, w, t_2) &= t_1 \\
  \mathsf{replace}(\mvty {t_1} , w, t_2) &= \mvty {\mathsf{replace}(t_1, w, t_2)} \\
  \mathsf{replace}(\G, w, t_2) &= \{ x \mapsto \mathsf{replace}(\tilde t_1, w, t_2) : (x \mapsto \tilde t_1) \in \G \} \\
  \\
  \mathsf{update}(\G, w, t_2) &= \mathsf{replace}(\writee(\G, w, t_2), w, t_2)
\end{align*}

\chapter*{Evaluation}

\newcommand{\config}[2]{\langle \ #1 \ , \ #2 \ \rangle}

\begin{align*}
  \ell &\cceq \ell_x &\text{(locations, $\mathcal L$)} \\
  v
  &\cceq \unitt \alt n \alt \ell
  &\text{(values, $\Val$)} \\
  \tilde v
  &\cceq \bot \alt v
  &\text{(partial values, $\tilde \Val$)} \\
  \\
  S &\in \Loc \mapsto \tilde \Val &\text{(store)} \\
  \\
  S &\ent w \rightsquigarrow \ell &\text{(place locations)} \\
  S &\ent e \evl v \tne S &\text{(expressions)} \\
  S &\ent s \tne S &\text{(statements)} \\
  S &\ent p \evl v \tne S &\text{(programs)}
\end{align*}

\begin{mathpar}
  \inferrule*[right=locVar]{
  }{
    S \ent x \rightsquigarrow \ell_x
  }
  \and
  \inferrule*[right=locDeref]{
    S \ent w \rightsquigarrow \ell_x
    \and
    (\ell_x \mapsto \ell_y) \in S
  }{
    S \ent \dref w \rightsquigarrow \ell_y
  }
  \and
  \inferrule*[right=unit]{
  }{
    S \ent \unt \evl \unt \tne S
  }
  \and
  \inferrule*[right=int]{
    n \in \Z
  }{
    S \ent n \evl n \tne S
  }
  \and
  \inferrule*[right=placeCopy]{
    S \ent w \rightsquigarrow \ell_x
    \and
    (\ell_x \mapsto v) \in S
  }{
    S \ent \copye w \evl v \tne S
  }
  \and
  \inferrule*[right=move]{
    S \ent w \rightsquigarrow \ell_x
    \and
    (\ell_x \mapsto v) \in S
  }{
    S \ent w \evl v \tne S[\ell_x \mapsto \bot]
  }
  \and
  \inferrule*[right=ref]{
    S \ent w \rightsquigarrow \ell_x
  }{
    S \ent \brw w \evl \ell_x \tne S
  }
  \and
  \inferrule*[right=mutRef]{
    S \ent w \rightsquigarrow \ell_x
  }{
    S \ent \mbrw w \evl \ell_x \tne S
  }
  \and
  \inferrule*[right=assign]{
    S \ent w \rightsquigarrow \ell_x
    \and
    (\ell_x \mapsto \tilde v) \in S_1
    \and
    S_1 \ent e \evl v \tne S_2
  }{
    S_1 \ent w \eqe e \evl \unt \tne S_2[\ell_x \mapsto v]
  }
  \and
  \inferrule*[right=let]{
    S_1 \ent e \evl v \tne S_2
  }{
    S_1 \ent \lete x \eqe e \tne S_2[\ell_x \mapsto v]
  }
  \and
  \inferrule*[right=letMut]{
    S_1 \ent e \evl v \tne S_2
  }{
    S_1 \ent \lete \mute x \eqe e \tne S_2[\ell_x \mapsto v]
  }
  \and
  \inferrule*[right=exprStmt]{
    S_1 \ent e \evl v \tne S_2
  }{
    S_1 \ent e \tne S_2
  }
  \and
  \inferrule*[right=prog]{
    S_1 \ent s \tne S_2
    \and
    S_2 \ent p \evl v \tne S_3
  }{
    S_1 \ent s \semi p \evl v \tne S_3
  }
\end{mathpar}
