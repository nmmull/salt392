\chapter*{Syntax}

\begin{align*}
  x
  &
  &\text{(variables, $\V$)} \\
  n
  &
  &\text{(integers, $\Z$)} \\
  w &\cceq x \alt \dref w &\text{(place expression, $\mathcal W$)} \\
  e
  &\cceq
  \unitt
  \alt n
  \alt w
  \alt \brw w
  \alt \mbrw w
  \alt w \eqe e
  &\text{(expressions, $\E$)} \\
  s
  &\cceq
  e
  \alt \lete x \eqe e
  \alt \lete \mute x \eqe e
  &\text{(statements, $\mathcal S$)} \\
  p
  &\cceq
  e
  \alt s \semi p
  &\text{(programs, $\cP$)}
\end{align*}

\chapter*{Typing}

\begin{align*}
  t
  &\cceq
  \unitty
  \alt \intty
  \alt \brw w
  \alt \mbrw w
  &\text{(types, $\T$)} \\
  \tilde t
  &\cceq
  \mvty t
  \alt t
  &\text{(partial types, $\tilde \T$)} \\
  m &\cceq \mathsf{imm} \alt \mathsf{mut} &\text{(mutability)} \\
  \\
  \G &\in \V \mapsto \T \times \{\mathsf{imm}, \mathsf{mut}\} & \text{(contexts)}\\
  \\
  \G &\ent w : \tilde t^m & \text{(place expressions)} \\
  \G &\ent \mathsf{writable}(w) &\text{(writability)} \\
  \G &\ent \tilde t \approx \tilde t &\text{(type compatibility)} \\
  \G &\ent e : t \tne \G &\text{(expressions)} \\
  \G &\ent s \tne \G &\text{(statments)} \\
  \G &\ent p : t \tne \G &\text{(programs)}
\end{align*}

\begin{align*}
  \mathsf{copy}(t) \equiv \nexists w . t = \mbrw w
\end{align*}

\begin{align*}
  \writee(\G, x, t) &= \G[x \mapsto t] \\
  \writee(\G, \code*^{k + 1}x, t) &= \writee(\G, \code*^k w, t) \quad \text{where } (x \mapsto \mbrw w^m) \in \G
\end{align*}

\chapter*{Evaluation}

\newcommand{\config}[2]{\langle \ #1 \ , \ #2 \ \rangle}

\begin{align*}
  \ell &\cceq \ell_x &\text{(locations, $\mathcal L$)} \\
  v
  &\cceq \unitt \alt n \alt \ell^m
  &\text{(values, $\Val$)} \\
  \tilde v
  &\cceq \bot \alt v
  &\text{(partial values, $\tilde \Val$)} \\
  \\
  S &\in \Loc \mapsto \tilde \Val \times \{\mathsf{imm}, \mathsf{mut}\} &\text{(store)} \\
  \\
  \config S e &\Downarrow \config S v &\text{(expressions)} \\
  \config S s &\Downarrow S &\text{(statements)} \\
  \config S p &\Downarrow \config S v &\text{(programs)}
\end{align*}

\begin{align*}
  \loc S x &= \ell_x \\
  \loc S {\dref w} &= \ell_x \quad \text{where } S(\loc S w) = \ell_x^{m}
\end{align*}
\bigskip

\begin{center}
  \AxiomC{}
  \RightLabel{(unit)}
  \UnaryInfC{$\config S \unitt \evl \config S \unitt$}
  \DisplayProof
\end{center}

\begin{center}
  \AxiomC{$n \in \mathbb Z$}
  \RightLabel{(int)}
  \UnaryInfC{$\config S n \evl \config S n$}
  \DisplayProof
\end{center}

\begin{center}
  \AxiomC{$S(\loc S w) \neq \ell_x^\mathsf{mut}$}
  \RightLabel{(place-copy)}
  \UnaryInfC{$\config S w \evl \config S {S(\loc S w)}$}
  \DisplayProof
\end{center}

\begin{center}
  \AxiomC{$S(\loc S w) = \ell_x^\mathsf{mut}$}
  \RightLabel{(place-move)}
  \UnaryInfC{$\config S w \evl \config {S[\loc S w \mapsto \bot]} {S(\loc S w)}$}
  \DisplayProof
\end{center}
